Google Sheets Setup Guide
This tool saves all guest data to a Google Sheet that also powers the volunteer scanner. Follow these steps once before your event.
1

Create your Google Sheet

Go to sheets.google.com and create a new spreadsheet. Name it Passover 2025 Meal Tracker. Add these exact column headers in row 1:

GuestID | FirstName | LastName | Contact | RegType | MealPlan | Days | RegisteredAt | Notes | B_Mar31 | L_Mar31 | B_Apr1 | L_Apr1 | B_Apr2 | L_Apr2 | B_Apr3 | L_Apr3 | B_Apr4 | L_Apr4 | B_Apr5 | L_Apr5 | B_Apr6 | L_Apr6 | B_Apr7 | L_Apr7 | B_Apr8 | L_Apr8

2

Deploy a Google Apps Script Web App

In your Sheet, go to Extensions → Apps Script. Paste the script from the code block below. Click Deploy → New Deployment → Web App. Set "Who has access" to Anyone. Copy the Web App URL.

3

Enter your Web App URL

Paste your Apps Script URL below. This connects the registration tool and the scanner app to your live Google Sheet.

4

Apps Script code (copy this)

Paste this into your Apps Script editor:

const SHEET_NAME = 'Sheet1';

function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const sheet = ss.getSheetByName(SHEET_NAME);

    if (data.action === 'register') {
      const row = buildRow(data.guest);
      sheet.appendRow(row);
      return ok({guestId: data.guest.id});
    }

    if (data.action === 'scan') {
      const rows = sheet.getDataRange().getValues();
      const headers = rows[0];
      const colKey = data.mealCol; // e.g. "B_Apr1"
      const colIdx = headers.indexOf(colKey);
      if (colIdx < 0) return err('Unknown meal column');
      for (let i = 1; i < rows.length; i++) {
        if (rows[i][0] === data.guestId) {
          const cell = sheet.getRange(i+1, colIdx+1);
          if (cell.getValue() === 'USED') return ok({status:'already_used', name: rows[i][1]+' '+rows[i][2]});
          cell.setValue('USED');
          return ok({status:'ok', name: rows[i][1]+' '+rows[i][2], plan: rows[i][5]});
        }
      }
      return err('Guest not found');
    }

    if (data.action === 'lookup') {
      const rows = sheet.getDataRange().getValues();
      for (let i = 1; i < rows.length; i++) {
        if (rows[i][0] === data.guestId) {
          return ok({found:true, guest: rowToObj(rows[0], rows[i])});
        }
      }
      return ok({found:false});
    }

  } catch(e) { return err(e.toString()); }
}

function doGet(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(SHEET_NAME);
  const rows = sheet.getDataRange().getValues();
  const headers = rows[0];
  const guests = rows.slice(1).map(r => rowToObj(headers, r));
  return ContentService.createTextOutput(JSON.stringify({guests})).setMimeType(ContentService.MimeType.JSON);
}

function buildRow(g) {
  const base = [g.id, g.firstName, g.lastName, g.contact||'', g.regType, g.mealPlan, (g.days||[]).join(','), g.registeredAt, g.notes||''];
  const mealCols = new Array(16).fill('');
  return [...base, ...mealCols];
}

function rowToObj(headers, row) {
  const obj = {};
  headers.forEach((h,i) => obj[h] = row[i]);
  return obj;
}

function ok(d) { return ContentService.createTextOutput(JSON.stringify({ok:true,...d})).setMimeType(ContentService.MimeType.JSON); }
function err(msg) { return ContentService.createTextOutput(JSON.stringify({ok:false,error:msg})).setMimeType(ContentService.MimeType.JSON); }
