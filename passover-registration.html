const SN = 'Sheet1';
function doPost(e) {
  try {
    const d = JSON.parse(e.postData.contents);
    const s = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SN);
    if (d.action==='register') {
      const g=d.guest;
      s.appendRow([g.id,g.firstName,g.lastName,g.email||'',g.phone||'',g.tabernacle||'',g.ageGroup||'',g.paymentType||'',g.balanceNotes||'','pending',g.registeredAt,g.notes||'',...new Array(16).fill('')]);
      return R({ok:true});
    }
    if (d.action==='checkin') {
      const rows=s.getDataRange().getValues(), hdr=rows[0];
      for(let i=1;i{const o={};hdr.forEach((h,i)=>o[h]=r[i]);return o;})})).setMimeType(ContentService.MimeType.JSON);
}
function R(d){return ContentService.createTextOutput(JSON.stringify(d)).setMimeType(ContentService.MimeType.JSON);}
