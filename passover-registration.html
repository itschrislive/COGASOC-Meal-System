const SHEET_NAME = 'Sheet1';
function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const sheet = ss.getSheetByName(SHEET_NAME);
    if (data.action === 'register') { sheet.appendRow(buildRow(data.guest)); return ok({guestId: data.guest.id}); }
    if (data.action === 'scan') {
      const rows = sheet.getDataRange().getValues();

        if (rows[i][0] === data.guestId) {
          const cell = sheet.getRange(i+1, colIdx+1);

function buildRow(g) {
  return [g.id,g.firstName,g.lastName,g.contact||'',g.regType,g.mealPlan,(g.days||[]).join(','),g.registeredAt,g.notes||'',...new Array(16).fill('')];
